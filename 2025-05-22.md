## isAuthApiError
```tsx
catch (err) {
	console.error('Signup error:', err);

	if (isAuthApiError(err)) {
		setError(`認証エラー: ${err.message}`);
	} else if (err instanceof Error) {
		setError(err.message);
	} else {
		setError('登録に失敗しました');
	}
}
```
